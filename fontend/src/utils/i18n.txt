// Langue par dÃ©faut
let currentLang: string = "fr";

interface I18nInstance {
  setLanguage: (lang: string) => void;
  t: (key: string, params?: Record<string, any>) => string;
  readonly currentLanguage: string;
  getTranslations: () => Record<string, string>;
  init: () => string;
}

export const i18n: I18nInstance = {
  // Changer la langue
  setLanguage(lang: string): void {
    if (translations[lang]) {
      currentLang = lang;
      localStorage.setItem("preferred_language", lang);
      document.documentElement.lang = lang;
    }
  },

  // Obtenir la traduction
  t(key: string, params: Record<string, any> = {}): string {
    let translation =
      translations[currentLang]?.[key] || translations["fr"]?.[key] || key;

    // Remplacer les paramÃ¨tres dynamiques
    Object.keys(params).forEach((param) => {
      translation = translation.replace(`{{${param}}}`, params[param]);
    });

    return translation;
  },

  // Obtenir la langue actuelle
  get currentLanguage(): string {
    return currentLang;
  },

  // Obtenir toutes les traductions (pour le debug)
  getTranslations(): Record<string, string> {
    return translations[currentLang];
  },

  // Initialiser depuis le localStorage ou le navigateur
  init(): string {
    const savedLang = localStorage.getItem("preferred_language");
    const browserLang = navigator.language.split("-")[0];

    if (savedLang && translations[savedLang]) {
      currentLang = savedLang;
    } else if (translations[browserLang]) {
      currentLang = browserLang;
    }

    document.documentElement.lang = currentLang;
    return currentLang;
  },
